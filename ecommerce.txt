CREATE database ECOMMERCE;
USE ECOMMERCE;
CREATE TABLE  USER(
FNAME VARCHAR(20),
LNAME VARCHAR(20),
PHNO INT (10), 
EMAIL VARCHAR (20)
);
CREATE TABLE PRODUCTS(
PRODUCT_NAME VARCHAR(20),
PRODUCT_ID INT(5),
PRODUCT_PRICE INT(10),
PRODUCT_DESCRIPTION VARCHAR(30)
);
CREATE TABLE CART(
PRODUCT_NAME VARCHAR(10),
PRODUCT_ID INT(5),
PRODUCT_DESCRIPTION VARCHAR(30)
);
